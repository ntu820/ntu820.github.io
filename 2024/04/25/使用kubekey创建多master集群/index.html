使用kubekey创建多master集群

## 使用kubekey创建多master集群

1. 架构

三个主节点，五个工作节点，两个用于master负载均衡的节点，以及一个VIP，两个用于traefik负载均衡的节点，以及一个traefik的VIP。

4. 配置负载均衡

4.1 配置haproxy

# 安装keepalived和haproxy，两台机器都需要安装
yum install keepalived haproxy psmisc -y

# 配置haproxy，两台机器都配置相同
vi /etc/haproxy/haproxy.cfg

global
    log /dev/log  local0 warning
    chroot        /var/lib/haproxy
    pidfile       /var/run/haproxy.pid
    maxconn       4000
    user          haproxy
    group         haproxy
    daemon
   
    stats socket /var/lib/haproxy/stats
   
defaults
    log global
    option  httplog
    option  dontlognull
    timeout connect 5000
    timeout client 50000
    timeout server 50000
   
frontend kube-apiserver
    bind *:6443
    mode tcp
    option tcplog
    default_backend kube-apiserver
   
backend kube-apiserver
    mode tcp
    option tcplog
    option tcp-check
    balance roundrobin
    default-server inter 10s downinter 5s rise 2 fall 2 slowstart 60s maxconn 250 maxqueue 256 weight 100
    server kube-apiserver-1 10.16.214.21:6443 check
    server kube-apiserver-2 10.16.214.22:6443 check
    server kube-apiserver-3 10.16.214.23:6443 check

# 启动
systemctl restart haproxy
# 自启动
systemctl enable haproxy

4.2 配置keepalived

# 配置keepalived
vi /etc/keepalived/keepalived.conf

# 主节点
global_defs {
  notification_email {
  }
  router_id LVS_DEVEL
  vrrp_skip_check_adv_addr
  vrrp_garp_interval 0
  vrrp_gna_interval 0
}
   
vrrp_script chk_haproxy {
  script "killall -0 haproxy"
  interval 2
  weight 2
}
   
vrrp_instance haproxy-vip {
  state BACKUP
  priority 100
  interface eth0 # 当前网卡名称
  virtual_router_id 60
  advert_int 1
  authentication {
    auth_type PASS
    auth_pass 1111
  }
  unicast_src_ip 10.16.214.31 # 当前主机IP 
  unicast_peer {
    10.16.214.32 # 另一台主机IP
  }
   
  virtual_ipaddress {
    10.16.214.30/24 # VIP
  }
   
  track_script {
    chk_haproxy
  }
}

# 副节点
