Kubernetes运维处理

1. 手动续期k8s证书

使用kubeadm搭建的k8s集群默认的kubectl只有一年有效期。

步骤

  1. 备份当前k8s证书

    cp -r /etc/kubernetes /etc/kubernetes.bak

  2. 查看过期时间

    kubeadm alpha certs check-expiration

  3. 更新证书

    kubeadm alpha certs renew all

systemctl restart kubelet

cp -a /etc/kubernetes/admin.conf /root/.kube/config

  4. 需要更新的文件

    └─etc
    └─kubernetes
        │  admin.conf
        │  config
        │  controller-manager.conf
        │  scheduler.conf
        │  
        ├─etcd
        │   healthcheck-client.crt
        │   healthcheck-client.key
        │   peer.crt
        │   peer.key
        │   server.crt
        │   server.key
        │      
        └─pki
            apiserver-etcd-client.crt
            apiserver-etcd-client.key
            apiserver-kubelet-client.crt
            apiserver-kubelet-client.key
            apiserver.crt
            apiserver.key
            front-proxy-client.crt
            front-proxy-client.key

  5. 自动化脚本

    kubeadm alpha certs renew all
docker ps -af 'name=k8s_POD_(kube-apiserver|kube-controller-manager|kube-scheduler|etcd)-*' -q | xargs docker restart
cp /etc/kubernetes/admin.conf /root/.kube/config
kubeadm alpha certs check-expiration

2. 修改源码，证书调整为100年

  1. 下载源码，解压

    wget https://github.com/kubernetes/kubernetes/archive/refs/tags/v1.19.8.zip
unzip kubernetes-v1.19.8.zip
cd kubernetes

  2. 修改ca证书时间

    vi staging/src/k8s.io/client-go/util/cert/cert.go
    func NewSelfSignedCACert(cfg Config, key crypto.Signer) (*x509.Certificate, error) {
        now := time.Now()
        tmpl := x509.Certificate{
                SerialNumber: new(big.Int).SetInt64(0),
                Subject: pkix.Name{
                        CommonName:   cfg.CommonName,
                        Organization: cfg.Organization,
                },
                NotBefore:             now.UTC(),
                // NotAfter:              now.Add(duration365d * 10).UTC(), 原本为10年
                NotAfter:              now.Add(duration365d * 100).UTC(), // 调整到100年
                KeyUsage:              x509.KeyUsageKeyEncipherment | x509.KeyUsageDigitalSignature | x509.KeyUsageCertSign,
                BasicConstraintsValid: true,
                IsCA:                  true,
        }

        certDERBytes, err := x509.CreateCertificate(cryptorand.Reader, &tmpl, &tmpl, key.Public(), key)
        if err != nil {
                return nil, err
        }
        return x509.ParseCertificate(certDERBytes)
}

  4. 重新编译源码

    1. 安装go环境

      # 查询当前kubernetes所使用的go版本
cat ./build/build-image/cross/VERSION  # v1.19.8使用 v1.15.8-legacy-1的go版本，直接下载v1.15.8即可。
wget https://dl.google.com/go/go1.15.8.linux-amd64.tar.gz  # 下载其他版本，更改版本号
tar -zxvf go1.15.8.linux-amd64.tar.gz  -C /usr/local

    2. 配置go

      vi /etc/profile
# 写入以下内容
export GO_HOME=/usr/local/go
export PATH=$PATH:$GO_HOME/bin
# 生效
source /etc/profile
# 验证
go version  # go version go1.15.8 linux/amd64

    3. 安装编译所需组件

      yum install -y gcc make rsync jq

    4. 重新编译

      # 重新编译kubeadm
make all WHAT=cmd/kubeadm GOFLAGS=-v

# 编译kubelet
make all WHAT=cmd/kubelet GOFLAGS=-v

# 编译kubectl
make all WHAT=cmd/kubectl GOFLAGS=-v

# 编译完的kubeadm在 _output/bin/kubeadm 目录下，其中bin是使用了软连接
# 真实路径是_output/local/bin/linux/amd64/kubeadm

  5. 执行证书更换

    1. 备份文件

mkdir -p /etc/kubernetes/back &&  cp *.conf   /etc/kubernetes/back

    2. 按照上面方法进行证书更新

  6. 目前存在的问题

    替换后可使用命令kubeadm alpha certs renew all续期证书，但无法更改ca证书，集群需重建才可以；
    后续需继续研究，如何在已经运行的集群中，并且保证不中断业务的情况下续期ca及其他相关证书；

3. 批量处理pod

# 查询所有不正常的pod
kubectl get pod -A|grep [Evicted]

# 删除命名空间下不正常的pod
kubectl get pod -n [NS]|grep '[Evicted]'|awk '{print$1}'|xargs kubectl delete pod -n [NS]

自动处理不正常的pod

vi opt.sh

# opt.sh的内容
for i in $(kubectl get pod -A|grep [Evicted]|awk '!a[$1]++{print}'|awk '{print$1}')
do
kubectl get pod -n $i|grep '[Evicted]'|awk '{print$1}'|xargs kubectl delete pod -n $i
done

强制删除pod

kubectl delete pod [pod name] --force --grace-period=0 -n [namespace]

强制删除命名空间

kubectl delete namespace NAMESPACENAME --force --grace-period=0

4. 产生k8s管理源密钥

# 产生k8s管理员密钥
adm_account="k8s-dash-admin"
kubectl create serviceaccount ${adm_account} -n kube-system
kubectl create clusterrolebinding ${adm_account} --clusterrole=cluster-admin --serviceaccount=kube-system:${adm_account}
kubectl -n kube-system describe secrets $(kubectl -n kube-system get secret | grep ${adm_account} | awk '{print $1}')
