nacos集群安装

1. 在线安装mysql5.7

  • 安装步骤
# 获取mysql yum源
yum install -y wget && wget https://dev.mysql.com/get/mysql57-community-release-el7-9.noarch.rpm
# 安装yum源
rpm -ivh mysql57-community-release-el7-9.noarch.rpm
# 安装mysql
yum install mysql-community-server mysql-community-libs-compat mysql-community-libs mysql-community-embedded-devel mysql-community-embedded-compat mysql-community-embedded mysql-community-devel mysql-community-common mysql-community-client -y
  • 启动
# 启动mysql
systemctl start mysqld
# 开机启动
systemctl enable mysqld
# 需要找到mysql初始化的密码，cat /var/log/mysqld.log|grep root
  • 运行mysql_secure_installation进行一些初始化操作，主要是对更改密码、是否运行远程等

rpm包安装mysql5.7

  • 下载rpm包
https://dev.mysql.com/get/Downloads/MySQL-5.7/mysql-community-common-5.7.23-1.el7.x86_64.rpm
https://dev.mysql.com/get/Downloads/MySQL-5.7/mysql-community-libs-5.7.23-1.el7.x86_64.rpm
https://dev.mysql.com/get/Downloads/MySQL-5.7/mysql-community-client-5.7.23-1.el7.x86_64.rpm
https://dev.mysql.com/get/Downloads/MySQL-5.7/mysql-community-server-5.7.23-1.el7.x86_64.rpm
https://dev.mysql.com/get/Downloads/MySQL-5.7/mysql-community-devel-5.7.23-1.el7.x86_64.rpm
  • 安装mysql
# 下载自带的mariadb
yum remove mariadb mariadb-server mariadb-devel mariadb-libs
# 安装
rpm -ivh mysql-community-common-5.7.23-1.el7.x86_64.rpm 
rpm -ivh mysql-community-libs-5.7.23-1.el7.x86_64.rpm 
rpm -ivh mysql-community-client-5.7.23-1.el7.x86_64.rpm 
rpm -ivh mysql-community-server-5.7.23-1.el7.x86_64.rpm 
rpm -ivh mysql-community-devel-5.7.23-1.el7.x86_64.rpm
  • 新建主库文件夹
mkdir -p /data/mysql/{data,log,pid,binary}
groupadd mysql
useradd -r -g mysql mysql
chown -R mysql.mysql  /data
  • 配置主数据库，vi /etc/my.cnf
[mysqld]
datadir=/data/mysql/data
socket=/data/mysql/data/mysql.sock
 
log-error=/data/mysql/log/mysqld.log
pid-file=/data/mysql/pid/mysqld.pid
 
user=mysql
server-id=1
port=3306
 
##要给从机同步的库
#binlog-do-db=
 
##不给从机同步的库(多个写多行)
binlog-ignore-db=mysql
binlog-ignore-db=information_schema
binlog-ignore-db=performance_schema
binlog-ignore-db=sys
 
##开启二进制日志
log-bin=/data/mysql/binary/mysql1-bin
 
##自动清理 7 天前的log文件,可根据需要修改
expire_logs_days=7
 
[client]
socket=/data/mysql/data/mysql.sock
  • 新建从库文件夹
mkdir -p /data/mysql/{data,log,pid}
groupadd mysql
useradd -r -g mysql mysql
chown -R mysql.mysql  /data
  • 配置从数据库
[mysqld]
datadir=/data/mysql/data
socket=/data/mysql/data/mysql.sock
 
# Disabling symbolic-links is recommended to prevent assorted security risks
symbolic-links=0
 
log-error=/data/mysql/log/mysqld.log
pid-file=/data/mysql/pid/mysqld.pid
 
user=mysql
server-id=2
port=3306
 
##从库上的参数
read_only = 1
master_info_repository=TABLE
relay_log_info_repository=TABLE
#relay_log_recovery=1   #从机禁止写
#super_read_only=1      #从机禁止写
 
[client]
socket=/data/mysql/data/mysql.sock
  • 启动数据库并找出root密码
systemctl start mysqld

# 查询密码
cat /data/mysql/log/mysqld.log |grep root
  • 登录并修改密码
mysql -uroot -p

# 修改密码
mysql-> ALTER USER root@localhost identified by 'admin123.X';

配置主从

  • 在主服务器上授权从服务器复制帐号
mysql -uroot -p
mysql-> grant replication slave on *.* to slave@'192.168.%' identified by 'admin123.X';
mysql-> show master status;
  • 到从服务配置主数据库连接
mysql -uroot -p

mysql > stop slave;
mysql > change master to master_host='192.168.153.162',master_port=3306,master_user='slave',master_password='admin123.X',master_log_file='mysql1-bin.000093',master_log_pos=846;
mysql > start slave;
mysql > show slave status\G
# 必须看到两个YES
# Slave_IO_Running: Yes
# Slave_SQL_Running: Yes

验证

mysql > CREATE DATABASE test_ab default charset utf8;
mysql > CREATE TABLE test_ab.a1(id int(2),name varchar(20));
mysql > INSERT INTO test_ab.a1(id,name) VALUES(1,"测试1");
mysql> select * from test_ab.a1;

2. 安装Nacos1.3

  • 配置JDK、Maven
# 解压缩
tar -zxvf jdk-8u261-linux-x64.tar.gz
mv jdk1.8.0_261 /usr/local/bin/
tar -zxvf apache-maven-3.6.3-bin.tar.gz
mv apache-maven-3.6.3 /usr/local/bin/

# 配置环境变量
vi /etc/profile

# 在最后添加
export MAVEN_HOME=/usr/local/bin/apache-maven-3.6.3
export JAVA_HOME=/usr/local/bin/jdk1.8.0_261
export PATH=$PATH:$MAVEN_HOME/bin
export PATH=$PATH:$JAVA_HOME/bin

# 配置生效
source /etc/profile

# 验证生效
java -version
mvn -v
  • 为nacos创建数据库连接用户
create user nacos identified by 'nacos123.X';

grant all privileges on *.* to 'nacos'@'192.168.%' identified by 'nacos123.X' with grant option;

flush privileges;
  • 解压nacos，并移动到安装目录
tar -zxvf nacos-server-1.3.0.tar.gz
mv nacos /usr/local/
  • 初始化数据库脚本，找到 nacos-mysql.sql ，将其导入到数据库中
  • 配置文件，vi application.properties
spring.datasource.platform=mysql
db.num=1
db.url.0=jdbc:mysql://192.168.153.161:3306/nacos_config?characterEncoding=utf8&connectTimeout=1000&socketTimeout=3000&autoReconnect=true
db.user=nacos
db.password=nacos123.X
  • 把cluster.conf.example 改名成 cluster.conf ，并修改配置vi cluster.conf
#it is ip
10.0.0.1:8848
10.0.0.2:8848
10.0.0.3:8848
  • 分别部署到服务器
sh startup.sh
  • 验证，访问不同主机的地址，查看是否可以正常访问

3. 配置nginx反向代理

# 下载nginx安装包
wget https://nginx.org/download/nginx-1.18.0.tar.gz
# 安装依赖
yum -y install gcc zlib zlib-devel pcre-devel openssl openssl-devel
# 解压
mkdir -p /usr/local/nginx
tar -zxvf nginx-1.18.0.tar.gz
mv nginx-1.18.0 /usr/local/nginx/
# 执行configure
cd /usr/local/nginx/nginx-1.18.0
./configure
make
make install
  • 配置nginx反向代理
upstream nacos {
        server  192.168.153.161:8848;
        server  192.168.153.162:8848;
        server  192.168.153.163:8848;
    }

server {
        listen       80;
        server_name  192.168.153.163;

        location /nacos/ {
            proxy_pass http://nacos/nacos/;
        }
    }
  • 启动nginx
/usr/local/nginx/sbin/nginx -s reload

# 如果出现报错：nginx: [error] open() ＂/usr/local/nginx/logs/nginx.pid＂ failed
运行： `/usr/local/nginx/sbin/nginx -c /usr/local/nginx/conf/nginx.conf` 再次启动即可
