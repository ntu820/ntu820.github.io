Linux7.9安装Mysql-5.7.21

下载

解压到目录

# 解压
tar -zxvf mysql-5.7.21-linux-glibc2.12-x86_64.tar.gz
# 移动目录
mv mysql-5.7.21-linux-glibc2.12-x86_64 /usr/local/mysql

新增用户、组和数据目录

# 新增组
groupadd mysql
# 新增用户
useradd -r -g mysql mysql
# 创建目录
mkdir -p /data/mysql
# 分配权限
chown mysql:mysql -R /data/mysql

配置my.cnf

vi /etc/my.cnf

# 可以采用这个
[mysqld]
bind-address=0.0.0.0
port=3306
user=mysql
basedir=/usr/local/mysql
datadir=/data/mysql
socket=/tmp/mysql.sock
log-error=/data/mysql/mysql.err
pid-file=/data/mysql/mysql.pid
#character config
character_set_server=utf8mb4
symbolic-links=0
explicit_defaults_for_timestamp=true

# 也可以采用这个
[mysql]
# 设置mysql客户端默认字符集
default-character-set=utf8
socket=/tmp/mysql.sock
[mysqld]
#skip-name-resolve
#skip-grant-tables
#设置3306端口
port=3306
socket=/tmp/mysql.sock
# 设置mysql的真正的安装目录 （看自己的是在哪里)
basedir=/usr/local/mysql
# 设置mysql数据库的数据的存放目录 （看自己的是在哪里)
datadir=/data/mysql
# 允许最大连接数
max_connections=200
# 服务端使用的字符集默认为8比特编码的latin1字符雿
character-set-server=utf8
# 创建新表时将使用的默认存储引擿
default-storage-engine=INNODB
#lower_case_table_name=1
max_allowed_packet=16M

初始化数据库

cd /usr/local/mysql/bin
./mysqld --defaults-file=/etc/my.cnf --basedir=/usr/local/mysql/ --datadir=/data/mysql/ --user=mysql --initialize
# 成功以后，查看密码
cat /data/mysql/mysql.err

启动mysql

cp /usr/local/mysql/support-files/mysql.server /etc/init.d/mysql
service start mysql

修改密码

mysql -uroot -p
# 输入刚刚获得的密码
# 修改密码
SET PASSWORD = PASSWORD('root');
ALTER USER 'root'@'localhost' PASSWORD EXPIRE NEVER;
FLUSH PRIVILEGES;

新增软连接

# 不用每次都进入mysql目录运行命令
ln -s  /usr/local/mysql/bin/mysql    /usr/bin

mysql数据丢失恢复

当mysql由于某些原因突然无法启动，且没有做过备份的情况下，使用mysql的data目录进行数据恢复

  1. 修改原my.cnf文件，重新知道一个data目录

  2. 启动mysql，进入mysql命令行，看是否启动成功

  3. 停止mysql，否则无法备份

  4. 在新的data目录下，备份文件

    cp auto.cnf auto.cnf.bak
cp ib_buffer_pool ib_buffer_pool.bak
cp ibdata1 ibdata1.bak
cp ib_logfile0 ib_logfile0.bak
cp ib_logfile1 ib_logfile1.bak

  5. 将需要恢复的目录和文件进行打包

    tar -zcvf backup.tar.gz auto.cnf ib_buffer_pool ibdata1 ib_logfile0 ib_logfile1 [需要恢复的数据库目录]

  6. 将压缩包导入进新的data目录，并解压

  7. 重新启动mysql，查看数据是否恢复

  8. 重新在新的mysql上分配用户和权限
