ESB单机环境搭建

ESB介绍

Mule是一个企业服务总线(ESB)消息框架。它的主要特性包括：

  1. 基于J2EE1.4的企业消息总线(ESB)和消息代理(broker)。

  2. 可插入的连接性:比如Jms,jdbc,tcp,udp,multicast,http,servlet,smtp,pop3, file,xmpp。

  3. 支持任何传输之上的异步，同步和请求响应事件处理机制。

  4. 支持Axis或者Glue的Web Service。

  5. 灵活的部署结构[Topologies]包括Client/Server, P2P, ESB 和Enterprise Service Network。

  6. 与Spring 框架集成:可用作ESB 容器，也可以很容易的嵌入到Spring应用中。

  7. 使用基于SEDA处理模型的高度可伸缩的企业服务器。

  8. 强大的基于EIP模式的事件路由机制等。

    Mule发布最新版本1.1，这个发布包括集成了 JBI，对 BPEL的支持,还增加一些新的传输器(transport)Quartz,FTP,RMI与EJB等。

安装mysql8数据库

  1. 使用mysql-8.0.21-linux-glibc2.12-x86_64.tar.xz 安装包

    # 解压
tar -xJvf mysql-8.0.21-linux-glibc2.12-x86_64.tar.xz -C /esb
mv /esb/mysql-8.0.21-linux-glibc2.12-x86_64 /esb/mysql

  2. 创建mysql用户组和用户

    groupadd mysql
useradd -g mysql mysql

  3. 更改mysql目录权限

    chown -R mysql:mysql /esb/mysql

  4. 更改mysql配置文件，vi /etc/my.cnf

    [mysqld]
port = 3306
mysqlx_port = 33060
basedir = /esb/mysql
datadir = /esb/mysql/data
socket = /tmp/mysql.sock
mysqlx_socket = /tmp/mysqlx.sock
pid-file = /tmp/mysqld.pid
log-error = error.log
slow-query-log = 1
slow-query-log-file = slow.log
long_query_time = 0.2
log-bin = mysql-bin
max-binlog-size = 50M
log-bin-index = bin.index
relay-log = mysql-relay-bin
binlog_format = mixed
relay_log_recovery = 1
character-set-client-handshake = FALSE
character-set-server = utf8mb4
collation-server = utf8mb4_unicode_ci
init_connect = 'SET NAMES utf8mb4'
innodb_buffer_pool_size = 1G
join_buffer_size = 128M
sort_buffer_size = 2M
read_rnd_buffer_size = 2M
log_timestamps = SYSTEM
lower_case_table_names = 1
default_authentication_plugin = mysql_native_password
max_connections = 500
wait_timeout = 600
interactive_timeout = 600

  5. 设置mysql服务为系统服务

    cp -a /esb/mysql/support-files/mysql.server /etc/init.d/mysql
chmod +x /etc/rc.d/init.d/mysql
chkconfig --add mysql

  6. 配置全局环境变量

    echo 'export PATH=/esb/mysql/bin:/esb/mysql/lib:$PATH' >> /etc/profile
# 生效
source /etc/profile

  7. 初始化数据库

    mysqld --initialize --user=mysql --basedir=/esb/mysql --datadir=/esb/mysql/data

  8. 启动mysql

    systemctl start mysql

  9. 查看root账号密码

    cat /esb/mysql/data/error.log | grep -i password

  10. 登录mysql，修改root密码

    mysql -uroot -p
# 输入上步骤得到的密码
# 以下为mysql命令行
alter user 'root'@'localhost' identified by 'XXXX';
set global log_bin_trust_function_creators = 1;
commit;
exit;

创建ESB的数据库

  1. 创建数据库

    # 试用root账号进行登录
mysql -uroot -p
# 输入正确密码后，进入mysql命令行
create database zjesbtest character set utf8 collate utf8_general_ci;
# 查看是否成功
show databases;

  2. 创建远程登录用户，并授权esb数据库

    # 以下均为mysql命令行
create user esbtest@'%' identified with mysql_native_password by 'XXXXX';
grant all privileges on esbtest.* to esbtest@'%';
flush privileges;
exit;

安装JDK

  1. 安装

    mkdir /home/esb/java
tar -zxvf jdk-8u181-linux-x64.tar.gz -C /home/esb/java

  2. 配置

    vi /etc/profile

# set java environment
export JAVA_HOME=/home/esb/java/jdk1.8.0_181
export PATH=$JAVA_HOME/bin:$JAVA_HOME/jre/bin:$PATH
export CLASSPATH=$CLASSPATH:.:$JAVA_HOME/lib:$JAVA_HOME/jre/lib

# 生效
source /etc/profile

安装ActiveMq

  1. 安装

    mkdir -p /home/esb/activemq
tar -zxvf apache-activemq-5.15.5-bin.tar.gz -C /home/esb/activemq

  2. 配置

    1. 调优配置

      • 调整JVM大小

        cd /home/esb/activemq/apache-activemq-5.15.5/bin/linux-x86-64
vi wrapper.conf

# 调整
wrapper.java.initmemory=1024
wrapper.java.maxmemory=2048

    2. 账号配置

      • 设置mq的web管理员账密，esbmqmgr/esbmqmgr

        cd /home/esb/activemq/apache-activemq-5.15.5/conf/
vi jetty-realm.properties
# 注释如下两行，添加最后一行
#admin: admin, admin
#user: user, user
esbmqmgr: esbmqmgr, admin

      • 设置mq的代理访问账密，esbadminrw/esbadminrw

        cd /home/esb/activemq/apache-activemq-5.15.5/conf/

# 添加如下用户
vi users.properties
# 注释如下一行，添加最后两行
# admin=admin
esbadminrw=esbadminrw
system=manager

# 设置用户角色
vi groups.properties
# 注释如下一行，添加最后一行
#admins=admin
admins=esbadminrw,system

      • 队列权限配置

        cd /home/esb/activemq/apache-activemq-5.15.5/conf/

vi activemq.xml
# 在</persistenceAdapter>标签后<systemUsage>标签前添加如下类似内容
        <plugins>
  <loggingBrokerPlugin logProducerEvents="true" logConsumerEvents="true"/>
  <jaasAuthenticationPlugin configuration="activemq" />
  <authorizationPlugin>
    <map>
      <authorizationMap>
        <authorizationEntries>
          <authorizationEntry queue=">" read="admins" write="admins" admin="admins" />
          <authorizationEntry topic=">" read="admins" write="admins" admin="admins" />
        </authorizationEntries>
      </authorizationMap>
    </map>
  </authorizationPlugin>
</plugins>

  3. 启动MQ

    cd /home/esb/activemq/apache-activemq-5.15.5/bin
# 启动
./activemq start
# 停止
./activemq stop

  4. 开机启动MQ

    # 指定JAVA_HOME，否则无法
cd /home/esb/activemq/apache-activemq-5.15.5/bin
echo "JAVA_HOME=\"/home/esb/java/jdk1.8.0_181\"" >> env

cat > /lib/systemd/system/activemq.service << EOF
[Unit]
Description=ActiveMQ service
After=network.target
[Service]
Type=forking
WorkingDirectory=/home/esb/activemq/apache-activemq-5.15.5/data
ExecStart=/home/esb/activemq/apache-activemq-5.15.5/bin/activemq start
ExecStop=/home/esb/activemq/apache-activemq-5.15.5/bin/activemq stop
User=root
Group=root
[Install]
WantedBy=multi-user.target
EOF

# 开机自启动
systemctl enable activemq
# 启动
systemctl start activemq

安装Redis

  1. 安装

    # 安装gcc
yum install -y gcc-c++
# 解压
tar -zxvf redis-5.0.10.tar.gz -C /home/esb
mv /home/esb/redis-5.0.10 /home/esb/redis
# 编译安装
cd /home/esb/redis
make MALLOC=libc
# 注意PREFIX要大写
make install PREFIX=/home/esb/redis

  2. 更改配置

    cd /home/esb/redis
mkdir log
vi redis.conf

bind 0.0.0.0
daemonize yes
requirepass 123123
logfile "/home/esb/redis/log/redis.log"

  3. 开机自启redis

    cp /home/esb/redis/utils/redis_init_script /etc/init.d/redisd
vi /etc/init.d/redisd

REDISPORT=6379
EXEC=/home/esb/redis/bin/redis-server
CLIEXEC=/home/esb/redis/bin/redis-cli
PIDFILE=/var/run/redis_${REDISPORT}.pid
CONF="/home/esb/redis/bin/${REDISPORT}.conf"

chkconfig redisd on

service redisd start
service redisd stop

安装Mule Runtime

  1. 安装

    mkdir -p /home/esb/mule
tar -zxvf mule-standalone-3.9.0.tar.gz -C /home/esb/mule

  2. 配置

    1. 调优配置，cd /home/esb/mule/mule-standalone-3.9.0/conf，备份老的wrapper.conf，用新的wrapper.conf替换。

    2. Bar包替换，cd /home/esb/mule/mule-standalone-3.9.0/lib/opt

      备份如下5个jar包：
connection-pool-2.3.33.jar
grizzly-framework-2.3.33.jar
grizzly-http-2.3.33.jar
grizzly-http-server-2.3.33.jar
grizzly-websockets-2.3.33.jar
添加如下5个最新的jar包：
connection-pool-2.3.35.jar
grizzly-framework-2.3.35.jar
grizzly-http-2.3.35.jar
grizzly-http-server-2.3.35.jar
grizzly-websockets-2.3.35.jar
添加后为新增的jar包添加可执行权限
      mv connection-pool-2.3.33.jar connection-pool-2.3.33.jar.bak
mv grizzly-framework-2.3.33.jar grizzly-framework-2.3.33.jar.bak
mv grizzly-http-2.3.33.jar grizzly-http-2.3.33.jar.bak
mv grizzly-http-server-2.3.33.jar grizzly-http-server-2.3.33.jar.bak
mv grizzly-websockets-2.3.33.jar grizzly-websockets-2.3.33.jar.bak
chmod +x connection-pool-2.3.35.jar
chmod +x grizzly-framework-2.3.35.jar
chmod +x grizzly-http-2.3.35.jar
chmod +x grizzly-http-server-2.3.35.jar
chmod +x grizzly-websockets-2.3.35.jar

  3. 开机自启

    # 此种方法未能实现开机自启
cat > /lib/systemd/system/mule.service << EOF
[Unit]

After=network.target
[Service]

PIDFile=/tmp/mule.pid
ExecStart=/home/esb/mule/mule-standalone-3.9.0/bin/mule start
ExecStop=/home/esb/mule/mule-standalone-3.9.0/bin/mule stop
User=root
Group=root
[Install]
WantedBy=multi-user.target
EOF

ESB运维处理

  1. zipkin与activemq冲突

    Caused by: org.springframework.beans.factory.UnsatisfiedDependencyException: Error creating bean with name 'czJmsListenerContainerFactory' defined in class path resource [org/springblade/esblogmq/config/ActivemqConfig.class]: Unsatisfied dependency expressed through method 'czJmsListenerContainerFactory' parameter 0; nested exception is org.springframework.beans.factory.BeanNotOfRequiredTypeException: Bean named 'czConnectionFactory' is expected to be of type 'org.apache.activemq.ActiveMQConnectionFactory' but was actually of type 'org.springframework.cloud.sleuth.instrument.messaging.LazyTopicConnectionFactory'

    解决方案：去除链路追踪相关包，这是由于activemq中的jms连接和链路追踪的LazyTopicConnectionFactory冲突

    <dependency>
  <groupId>org.springblade</groupId>
  <artifactId>blade-starter-trace</artifactId>
</dependency>
